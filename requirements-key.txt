# Optional dependencies for the Key-detect feature (musical-key detection).
# Not required by the rest of song-sorter -- install only if you want to
# annotate index.json with a musical key for a pre-song pitch reference.
#
#   pip install -r requirements-key.txt
#
# Offline audio key detection (chromagram + Krumhansl-Schmuckler). librosa pulls
# in numpy/scipy/numba/soundfile. Decoding MP3s also needs an audio backend:
# a modern soundfile/libsndfile (>=1.1), or ffmpeg on your PATH.
librosa>=0.10,<0.12

# Online corroboration only (AcoustID fingerprint -> AcousticBrainz key).
# Needs the Chromaprint `fpcalc` binary on your PATH and a free AcoustID API key
# (https://acoustid.org/new-application). Everything works offline without this.
pyacoustid>=1.3.0
